# [53. 最大子数组和](https://leetcode-cn.com/problems/maximum-subarray/)

## code
```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
    const len = nums.length;
    const dp = new Array(len);

    dp[0] = nums[0];

    for(let i = 1;i < len;i++) {
        if(dp[i - 1] > 0) {
            dp[i] = dp[i - 1] + nums[i]
            continue;
        }
        dp[i] = nums[i];
    }

    return dp.reduce((memo,cur) => (cur > memo ? cur : memo),Number.MIN_SAFE_INTEGER)
};
```
