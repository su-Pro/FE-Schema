/**
 * @param {number[]} nums
 * @return {number}
 */
 var rob = function (nums) {
  const len = nums.length;
  if(len < 0) {
    return -1;
  }
  if(len === 1) {
    return nums[0]
  }
  const headRob = getRob(nums.slice(0,len - 1));
  const tailRob = getRob(nums.slice(1,len + 1));

  function getRob(source) {
    const len = source.length;
    const dp = new Array(len + 1);

    dp[0] = 0;
    dp[1] = source[0];

    for (let i = 2; i <= len; i++) {
      dp[i] = Math.max(dp[i - 1], dp[i - 2] + source[i - 1]);
    }
    return dp[len];
  }

  return Math.max(headRob, tailRob);
};
