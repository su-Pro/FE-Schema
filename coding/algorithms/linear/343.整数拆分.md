var integerBreak = function(n) {
    const dp = new Array(n + 1)


    dp[0] = 0;
    dp[1] = 1;
    dp[2] = 1;

    for(let i = 3;i <= n;i++) {
        let max = Number.MIN_SAFE_INTEGER;
        for(let j = 1;j < i;j++) {
            // (左大段 或者 不切) * 右小段(自成一派)
            max = Math.max(max,(i - j) * Math.max(dp[j],j));
        }
        dp[i] = max;
    }
    return dp[n];
};
