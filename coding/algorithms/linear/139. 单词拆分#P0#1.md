/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function(s, wordDict) {
    const len = s.length;
    let dp = new Array(len).fill(false);

    for(let i = 0;i < len;i++) {

        // case: 0
        if(wordDict.includes(s.substring(0,i + 1))) {
            dp[i] = true;
            continue;
        }

        for(let j = i;j >= 0;j--) {

            if(
                dp[j - 1] === true 
                && wordDict.includes(s.substring(j,i + 1))
            ) {
                dp[i] = true;
                break;
            }

        }
    }
    return dp[len - 1]
};
